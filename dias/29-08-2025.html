Resumo da aula

O professor comentou uma teoria envolvendo três valores inteiros, sem sinal, de 8 bits, que podem ser combinados para formar um pixel de uma imagem. Com três matrizes desses valores, conseguimos montar uma imagem colorida, relacionando com teoria de cores e sinais eletromagnéticos.

Exercício 5 – Número primo ou não

package main

import "fmt"

func main() {
    var x int

    fmt.Print("Digite um número inteiro: ")
    fmt.Scan(&x)

    if x < 2 {
        fmt.Println("Não é primo.")
        return
    }

    primo := true
    for i := 2; i*i <= x; i++ {
        if x%i == 0 {
            primo = false
            break
        }
    }

    if primo {
        fmt.Println("É primo.")
    } else {
        fmt.Println("Não é primo.")
    }
}

Exercício 13 – Moda de uma sentença numérica

package main

import "fmt"

func main() {
    var n int
    fmt.Print("Quantos números terá a sentença? ")
    fmt.Scan(&n)

    numeros := make([]int, n)
    for i := 0; i < n; i++ {
        fmt.Printf("Número %d: ", i+1)
        fmt.Scan(&numeros[i])
    }

    freq := make(map[int]int)
    maxFreq := 0
    moda := numeros[0]

    for _, v := range numeros {
        freq[v]++
        if freq[v] > maxFreq {
            maxFreq = freq[v]
            moda = v
        }
    }

    fmt.Println("Moda da sentença numérica:", moda)
}

Exercício 14 – Verificar se é palíndromo

package main

import (
    "bufio"
    "fmt"
    "os"
    "strings"
)

func normalizar(s string) string {
    s = strings.ToLower(s)
    s = strings.ReplaceAll(s, " ", "")
    return s
}

func main() {
    reader := bufio.NewReader(os.Stdin)
    fmt.Print("Digite uma sequência de caracteres: ")
    input, _ := reader.ReadString('\\n')

    s := normalizar(input)
    i, j := 0, len(s)-1
    palindromo := true

    for i < j {
        if s[i] != s[j] {
            palindromo = false
            break
        }
        i++
        j--
    }

    if palindromo {
        fmt.Println("É palíndromo.")
    } else {
        fmt.Println("Não é palíndromo.")
    }
}

Exercício 17 – Jogo de adivinhação

package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano())
    segredo := rand.Intn(100) + 1 // 1..100

    var palpite int
    tentativas := 0

    fmt.Println("Jogo de adivinhação! Tente descobrir o número de 1 a 100.")

    for {
        fmt.Print("Seu palpite: ")
        fmt.Scan(&palpite)
        tentativas++

        if palpite == segredo {
            fmt.Println("Acertou! Número de tentativas:", tentativas)
            break
        } else if palpite < segredo {
            fmt.Println("Tente um número maior.")
        } else {
            fmt.Println("Tente um número menor.")
        }
    }
}

Exercício 18 – Três matrizes → arquivo .jpg

package main

import (
    "image"
    "image/color"
    "image/jpeg"
    "os"
)

func main() {
    largura := 256
    altura := 256

    // Matrizes de 8 bits sem sinal (0–255).
    R := make([][]uint8, altura)
    G := make([][]uint8, altura)
    B := make([][]uint8, altura)

    for y := 0; y < altura; y++ {
        R[y] = make([]uint8, largura)
        G[y] = make([]uint8, largura)
        B[y] = make([]uint8, largura)
        for x := 0; x < largura; x++ {
            R[y][x] = uint8(x)      // gradiente horizontal
            G[y][x] = uint8(y)      // gradiente vertical
            B[y][x] = uint8(x ^ y)  // só pra variar
        }
    }

    img := image.NewRGBA(image.Rect(0, 0, largura, altura))

    for y := 0; y < altura; y++ {
        for x := 0; x < largura; x++ {
            c := color.RGBA{
                R: R[y][x],
                G: G[y][x],
                B: B[y][x],
                A: 255,
            }
            img.Set(x, y, c)
        }
    }

    arquivo, err := os.Create("saida.jpg")
    if err != nil {
        panic(err)
    }
    defer arquivo.Close()

    qualidade := &jpeg.Options{Quality: 90}
    err = jpeg.Encode(arquivo, img, qualidade)
    if err != nil {
        panic(err)
    }

    println("Imagem saida.jpg gerada com sucesso.")
}

Exercício 19 – Contar vogais e consoantes

package main

import (
    "bufio"
    "fmt"
    "os"
    "strings"
    "unicode"
)

func main() {
    reader := bufio.NewReader(os.Stdin)
    fmt.Print("Digite um texto: ")
    texto, _ := reader.ReadString('\\n')
    texto = strings.ToLower(texto)

    vogais := 0
    consoantes := 0

    for _, r := range texto {
        if unicode.IsLetter(r) {
            if strings.ContainsRune("aeiou", r) {
                vogais++
            } else {
                consoantes++
            }
        }
    }

    fmt.Println("Vogais:", vogais)
    fmt.Println("Consoantes:", consoantes)
}

Exercício 21 – Fatorial de N

package main

import "fmt"

func fatorial(n int) int {
    if n < 0 {
        return 0
    }
    resultado := 1
    for i := 2; i <= n; i++ {
        resultado *= i
    }
    return resultado
}

func main() {
    var n int
    fmt.Print("Digite um número inteiro: ")
    fmt.Scan(&n)

    fmt.Println("Fatorial:", fatorial(n))
}

Exercício 24 – MMC

package main

import "fmt"

func mdc(a, b int) int {
    for b != 0 {
        a, b = b, a%b
    }
    return a
}

func mmc(a, b int) int {
    return a * b / mdc(a, b)
}

func main() {
    var x, y int
    fmt.Print("Digite o primeiro número: ")
    fmt.Scan(&x)
    fmt.Print("Digite o segundo número: ")
    fmt.Scan(&y)

    fmt.Println("MMC:", mmc(x, y))
}

Exercício 25 – Média de dois ou mais números

package main

import "fmt"

func main() {
    var n int
    fmt.Print("Quantos números deseja calcular a média? ")
    fmt.Scan(&n)

    if n <= 0 {
        fmt.Println("Quantidade inválida.")
        return
    }

    for i := 0; i < n; i++ {

        fmt.Printf("Digite o %dº número: ", i+1)
        fmt.Scan(&valor)
        soma += valor
    }

    media := soma / float64(n)
    fmt.Println("Média:", media)
}
